```js
// –ë–∞–∑–æ–≤–∞—è —Ñ—É–Ω–∫—Ü–∏–æ–Ω–∞–ª—å–Ω–æ—Å—Ç—å set.map

// –¢—Ä–µ–∫–∏–Ω–≥ –¥–æ–±–∞–≤–ª–µ–Ω–∏—è —ç–ª–µ–º–µ–Ω—Ç–æ–≤
// –°–æ–±—ã—Ç–∏—è: sourceSet.dataAdd -> mappedSet.dataAdd

const deep2 = new deep()

// –°–ª–æ–∂–Ω—ã–µ –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏—è —Å –æ—Ç—Å–ª–µ–∂–∏–≤–∞–Ω–∏–µ–º —Å–æ—Å—Ç–æ—è–Ω–∏—è

// –ü—Ä–æ–≤–µ—Ä–∫–∞ —Å–æ—Å—Ç–æ—è–Ω–∏—è _state._mapValues (–≤–Ω—É—Ç—Ä–µ–Ω–Ω–∏–π Map –¥–ª—è —Ç—Ä–µ–∫–∏–Ω–≥–∞)

complexSet.add('c') // complexSet = { 'a', 'b', 'c' }
expect(upperCaseSet._data.has('C')).toBe(true) // upperCaseSet = { 'A', 'B', 'C' }
expect(upperCaseSet._state._mapValues.get('c')).toBe('C')

complexSet.delete('a') // complexSet = { 'b', 'c' }
expect(upperCaseSet._data.has('A')).toBe(false) // upperCaseSet = { 'B', 'C' }
expect(upperCaseSet._state._mapValues.has('a')).toBe(false) // –æ—á–∏—Å—Ç–∫–∞ map

// –û—Ç–∫–ª—é—á–µ–Ω–∏–µ —Ç—Ä–µ–∫–∏–Ω–≥–∞ - —Å–æ–±—ã—Ç–∏—è –¥–æ–ª–∂–Ω—ã –ø—Ä–µ–∫—Ä–∞—Ç–∏—Ç—å—Å—è
const disposer = complexSet.track(upperCaseSet)
disposer() // –æ—Ç–∫–ª—é—á–∞–µ–º —Ç—Ä–µ–∫–∏–Ω–≥
let eventFired = false
upperCaseSet.on(deep.events.dataAdd, () => { eventFired = true })
complexSet.add('d') // complexSet = { 'b', 'c', 'd' }
expect(upperCaseSet._data.has('D')).toBe(false) // upperCaseSet –æ—Å—Ç–∞–µ—Ç—Å—è { 'B', 'C' }
expect(eventFired).toBe(false) // —Å–æ–±—ã—Ç–∏—è –Ω–µ –¥–æ–ª–∂–Ω—ã –ø—Ä–æ–∏—Å—Ö–æ–¥–∏—Ç—å
```

#### –†–∞—Å–ø–∞–∫–æ–≤—â–∏–∫ —Ä–µ–ª–µ–π—à–µ–Ω–æ–≤ –≤ –º–Ω–æ–∂–µ—Å—Ç–≤–∞ –∞—Å—Å–æ—Ü–∏–∞—Ç–∏–≤–Ω—ã–π –º–µ—Ç–æ–¥ X.manyRelation(fieldName) üî¥

–ü—Ä–∏ –æ–±—Ä–∞—â–µ–Ω–∏–∏ –∫ –ª—é–±–æ–º—É —Ñ–∏–ª–¥—É –Ω–µ –≤–∞–∂–Ω–æ –æ–¥–∏–Ω–æ—á–Ω–æ–º—É –∏–ª–∏ –º–Ω–æ–∂–µ—Å—Ç–≤–µ–Ω–Ω–æ–º—É –≤–µ—Ä–Ω–µ—Ç –º–Ω–æ–∂–µ—Å—Ç–≤–µ–Ω–Ω—ã–π –ø–æ–¥–¥–µ—Ä–∂–∏–≤–∞—é—â–∏–π —Ç—Ä–µ–∫–∏–Ω–≥ —Å–µ—Ç.

**–¢–µ—Å—Ç–æ–≤—ã–µ —Å–ª—É—á–∞–∏:**
```js
// –ë–∞–∑–æ–≤–∞—è —Ñ—É–Ω–∫—Ü–∏–æ–Ω–∞–ª—å–Ω–æ—Å—Ç—å –æ–¥–∏–Ω–æ—á–Ω—ã—Ö –æ—Ç–Ω–æ—à–µ–Ω–∏–π
const X = new deep()
const a = new X()
const b = new X()

// a.type = X, –ø–æ—ç—Ç–æ–º—É a.manyRelation('type') = { X }

// X.typed = { a, b }, –ø–æ—ç—Ç–æ–º—É X.manyRelation('typed') = { a, b }

// –¢—Ä–µ–∫–∏–Ω–≥ –∏–∑–º–µ–Ω–µ–Ω–∏–π –æ–¥–∏–Ω–æ—á–Ω—ã—Ö –æ—Ç–Ω–æ—à–µ–Ω–∏–π
// –°–æ–±—ã—Ç–∏—è: –ø—Ä–∏ –∏–∑–º–µ–Ω–µ–Ω–∏–∏ a.type -> aTypeSet.dataChanged

// –°–æ–±—ã—Ç–∏—è: –ø—Ä–∏ —Å–æ–∑–¥–∞–Ω–∏–∏ –Ω–æ–≤–æ–≥–æ —ç–∫–∑–µ–º–ø–ª—è—Ä–∞ X -> XTypedSet.dataAdd
let addedToTyped = null
XTypedSet.on(deep.events.dataAdd, (element) => { addedToTyped = element._symbol })

expect(addedToTyped).toBe(c._id)

// –¢—Ä–µ–∫–∏–Ω–≥ —É–¥–∞–ª–µ–Ω–∏—è —Å–≤—è–∑–µ–π
// –°–æ–±—ã—Ç–∏—è: delete a.type -> —Å–≤—è–∑–∞–Ω–Ω—ã–µ –Ω–∞–±–æ—Ä—ã –æ–±–Ω–æ–≤–ª—è—é—Ç—Å—è
let deletedFromTyped = null
XTypedSet.on(deep.events.dataDelete, (element) => { deletedFromTyped = element._symbol })

delete a.type // —É–¥–∞–ª—è–µ–º —Å–≤—è–∑—å type —É a

// –†–∞–±–æ—Ç–∞ —Å value –æ—Ç–Ω–æ—à–µ–Ω–∏—è–º–∏

expect(str1ValuedSet._data).toEqual(new Set([a._id])) // { a }

// –ù–µ—Å—É—â–µ—Å—Ç–≤—É—é—â–∏–µ –ø–æ–ª—è –≤–æ–∑–≤—Ä–∞—â–∞—é—Ç –ø—É—Å—Ç—ã–µ –æ—Ç—Å–ª–µ–∂–∏–≤–∞–µ–º—ã–µ –Ω–∞–±–æ—Ä—ã

expect(emptySet._data.size).toBe(0) // { }
expect(emptySet instanceof deep.Deep).toBe(true)
expect(emptySet.type.is(deep.Set)).toBe(true)

// –ü—Ä–æ–≤–µ—Ä–∫–∞ —á—Ç–æ –ø—É—Å—Ç—ã–µ –Ω–∞–±–æ—Ä—ã —Ç–æ–∂–µ –ø–æ–¥–¥–µ—Ä–∂–∏–≤–∞—é—Ç —Ç—Ä–µ–∫–∏–Ω–≥ (–Ω–∞ —Å–ª—É—á–∞–π –±—É–¥—É—â–∏—Ö –∏–∑–º–µ–Ω–µ–Ω–∏–π)
let emptySetChanged = false
emptySet.on(deep.events.dataChanged, () => { emptySetChanged = true })
// –í –¥–∞–Ω–Ω–æ–º —Å–ª—É—á–∞–µ –∏–∑–º–µ–Ω–µ–Ω–∏–π –Ω–µ –¥–æ–ª–∂–Ω–æ –±—ã—Ç—å, –Ω–æ —Å—Ç—Ä—É–∫—Ç—É—Ä–∞ —Ç—Ä–µ–∫–∏–Ω–≥–∞ –¥–æ–ª–∂–Ω–∞ —Ä–∞–±–æ—Ç–∞—Ç—å
```

#### –†–∞—Å–ø–∞–∫–æ–≤—â–∏–∫ –∏–Ω–≤–µ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã—Ö –º–Ω–æ–∂–µ—Å—Ç–≤ deepSet.mapByField(fieldName) üî¥

–¢–∞–∫ –∫–∞–∫ –∑–∞–ø—Ä–æ—Å { type: X } –∏—â–µ—Ç "—Ç–µ—Ö —É –∫–æ–≥–æ —Ç–∏–ø X" –∞ { typed: a } –∏—â–µ—Ç —Ç–µ—Ö —É –∫–æ–≥–æ –µ—Å—Ç—å —ç–∫–∑–µ–º–ø–ª—è—Ä a, –∑–Ω–∞—á–∏—Ç –≤ –≤—ã—Ä–∞–∂–µ–Ω–∏—è—Ö –Ω—É–∂–Ω–∞ –∏–Ω–≤–µ—Ä—Ç–∞—Ü–∏—è.

**–ü–æ–¥—Ä–æ–±–Ω—ã–π –ø—Ä–∏–º–µ—Ä —Ä–∞–±–æ—Ç—ã mapByField:**
```js
// –ü—Ä–æ—Å—Ç–∞—è –∏–Ω–≤–µ—Ä—Ç–∞—Ü–∏—è
const X = new deep()
const a = new X() // a.type = X
const b = new X() // b.type = X

const typedSet = X.manyRelation('typed') // { a, b }
const invertedToType = typedSet.mapByField('type') // –∏—â–µ–º type –¥–ª—è –∫–∞–∂–¥–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞
// typedSet.map(element => element.manyRelation('type')) => [{ X }, { X }]
// deep.Or([{ X }, { X }]) => { X }
expect(invertedToType._data).toEqual(new Set([X._id])) // { X }

// –°–ª–æ–∂–Ω–∞—è –∏–Ω–≤–µ—Ä—Ç–∞—Ü–∏—è - –ø–æ–∫–∞–∑—ã–≤–∞–µ—Ç –∫–∞–∫ –ø–æ–ª—É—á–∞–µ—Ç—Å—è { e, q, o, p }
const R = new deep()    // –∫–æ—Ä–Ω–µ–≤–æ–π —Ç–∏–ø
const W = new R()       // W.type = R  
const Z = new R()       // Z.type = R
const e = new W()       // e.type = W
const q = new W()       // q.type = W  
const o = new Z()       // o.type = Z
const p = new Z()       // p.type = Z

// –ó–∞–ø—Ä–æ—Å: –Ω–∞–π—Ç–∏ –≤—Å–µ —ç–ª–µ–º–µ–Ω—Ç—ã, —á–µ–π —Ç–∏–ø –∏–º–µ–µ—Ç —Ç–∏–ø R
// –®–∞–≥ 1: deep.query({ type: R }) => { W, Z }
const typesWithR = new deep.Set(new Set([W._id, Z._id])) // { W, Z }

// –®–∞–≥ 2: typesWithR.mapByField('typed') 
// W.manyRelation('typed') => { e, q }
// Z.manyRelation('typed') => { o, p }  
// deep.Or([{ e, q }, { o, p }]) => { e, q, o, p }
const invertedResult = typesWithR.mapByField('typed')
expect(invertedResult._data).toEqual(new Set([e._id, q._id, o._id, p._id])) // { e, q, o, p }

// –¢—Ä–µ–∫–∏–Ω–≥ —á–µ—Ä–µ–∑ –∏–Ω–≤–µ—Ä—Ç–∞—Ü–∏—é - —Å–æ–∑–¥–∞–Ω–∏–µ –Ω–æ–≤–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞
// –°–æ–±—ã—Ç–∏—è: –Ω–æ–≤—ã–π —ç–ª–µ–º–µ–Ω—Ç -> –∏—Å—Ç–æ—á–Ω–∏–∫ –æ–±–Ω–æ–≤–ª—è–µ—Ç—Å—è -> —Ä–µ–∑—É–ª—å—Ç–∞—Ç –æ–±–Ω–æ–≤–ª—è–µ—Ç—Å—è
let addedToInverted = null
invertedResult.on(deep.events.dataAdd, (element) => { addedToInverted = element._symbol })

const r = new W() // —Å–æ–∑–¥–∞–µ–º –Ω–æ–≤—ã–π —ç–∫–∑–µ–º–ø–ª—è—Ä W
// r.type = W, –ø–æ—ç—Ç–æ–º—É W.typed —Ç–µ–ø–µ—Ä—å { e, q, r }
// invertedResult –¥–æ–ª–∂–Ω–æ —Å—Ç–∞—Ç—å { e, q, o, p, r }
expect(invertedResult._data.has(r._id)).toBe(true)
expect(addedToInverted).toBe(r._id)

// –¢—Ä–µ–∫–∏–Ω–≥ —á–µ—Ä–µ–∑ –∏–Ω–≤–µ—Ä—Ç–∞—Ü–∏—é - —É–¥–∞–ª–µ–Ω–∏–µ –∏—Å—Ç–æ—á–Ω–∏–∫–∞
let deletedFromInverted = null  
invertedResult.on(deep.events.dataDelete, (element) => { deletedFromInverted = element._symbol })

W.destroy() // —É–Ω–∏—á—Ç–æ–∂–∞–µ–º W
// –í—Å–µ —ç–∫–∑–µ–º–ø–ª—è—Ä—ã W (e, q, r) —Ç–µ—Ä—è—é—Ç —Å–≤—è–∑–∏
// invertedResult –¥–æ–ª–∂–Ω–æ —Å—Ç–∞—Ç—å { o, p }
expect(invertedResult._data).toEqual(new Set([o._id, p._id]))
expect(deletedFromInverted).toBeTruthy() // –æ–¥–∏–Ω –∏–∑ e, q, r

// –ü—É—Å—Ç—ã–µ —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã  
const emptySet = new deep.Set(new Set())
const emptyInverted = emptySet.mapByField('type')
expect(emptyInverted._data.size).toBe(0) // { }

// –û—Ç–∫–ª—é—á–µ–Ω–∏–µ —Ç—Ä–µ–∫–∏–Ω–≥–∞ –ø—Ä–µ–∫—Ä–∞—â–∞–µ—Ç —Å–æ–±—ã—Ç–∏—è
const newTypesSet = new deep.Set(new Set([W._id]))
const newInverted = newTypesSet.mapByField('typed')
const disposer = newTypesSet.track(newInverted) // —É—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º —Ç—Ä–µ–∫–∏–Ω–≥
disposer() // –æ—Ç–∫–ª—é—á–∞–µ–º —Ç—Ä–µ–∫–∏–Ω–≥

let eventAfterDispose = false
newInverted.on(deep.events.dataAdd, () => { eventAfterDispose = true })
const newElement = new W() // —Å–æ–∑–¥–∞–µ–º —ç–ª–µ–º–µ–Ω—Ç
expect(eventAfterDispose).toBe(false) // —Å–æ–±—ã—Ç–∏—è –Ω–µ –¥–æ–ª–∂–Ω—ã –ø—Ä–æ–∏—Å—Ö–æ–¥–∏—Ç—å
```

#### deep.queryField(fieldName, value) üî¥

–ö–æ–º–±–∏–Ω–∞—Ü–∏—è manyRelation –∏ mapByField –¥–ª—è –≤—ã–ø–æ–ª–Ω–µ–Ω–∏—è –ø–æ–∏—Å–∫–∞ –ø–æ –∫–æ–Ω–∫—Ä–µ—Ç–Ω–æ–º—É –ø–æ–ª—é.

**–¢–µ—Å—Ç–æ–≤—ã–µ —Å–ª—É—á–∞–∏:**

```js
const X = new deep()
const Y = new deep()
const a = new X() // a.type = X
const b = new X() // b.type = X

// –ü–æ–∏—Å–∫ —Å value –æ—Ç–Ω–æ—à–µ–Ω–∏—è–º–∏

const str2 = new deep.String('world')
a.value = str1 // a.value = str1
b.value = str2 // b.value = str2

const valuedStr1Result = deep.queryField('valued', str1)
expect(valuedStr1Result._data).toEqual(new Set([a._id])) // { a }

const valueStr1Result = deep.queryField('value', str1)  
expect(valueStr1Result._data).toEqual(new Set([a._id])) // { a }

// –í–ª–æ–∂–µ–Ω–Ω—ã–µ –æ–±—ä–µ–∫—Ç—ã –∑–∞–ø—Ä–æ—Å–æ–≤
const R = new deep()
const W = new R() // W.type = R
const Z = new R() // Z.type = R  
const e = new W() // e.type = W
const q = new W() // q.type = W

// deep.queryField('type', { type: R }) –¥–æ–ª–∂–Ω–æ –Ω–∞–π—Ç–∏ –≤—Å–µ —ç–ª–µ–º–µ–Ω—Ç—ã,
// —á–µ–π —Ç–∏–ø –∏–º–µ–µ—Ç —Ç–∏–ø R (—Ç.–µ. e, q, etc.)

expect(nestedResult._data).toEqual(new Set([e._id, q._id])) // { e, q }

// –¢—Ä–µ–∫–∏–Ω–≥ –≤ queryField - —Å–æ–∑–¥–∞–Ω–∏–µ –Ω–æ–≤—ã—Ö —ç–ª–µ–º–µ–Ω—Ç–æ–≤
let addedToQuery = null
typeXResult.on(deep.events.dataAdd, (element) => { addedToQuery = element._symbol })

const d = new X() // —Å–æ–∑–¥–∞–µ–º –Ω–æ–≤—ã–π —ç–∫–∑–µ–º–ø–ª—è—Ä X
expect(typeXResult._data.has(d._id)).toBe(true) // typeXResult = { a, b, d }
expect(addedToQuery).toBe(d._id)

// –¢—Ä–µ–∫–∏–Ω–≥ –≤ queryField - –∏–∑–º–µ–Ω–µ–Ω–∏–µ —Ç–∏–ø–æ–≤
let removedFromQuery = null
typeXResult.on(deep.events.dataDelete, (element) => { removedFromQuery = element._symbol })

expect(typeXResult._data.has(a._id)).toBe(false) // typeXResult = { b, d }

// –ù–µ—Å—É—â–µ—Å—Ç–≤—É—é—â–∏–µ –∑–Ω–∞—á–µ–Ω–∏—è –≤–æ–∑–≤—Ä–∞—â–∞—é—Ç –ø—É—Å—Ç—ã–µ –Ω–∞–±–æ—Ä—ã
```
